95  	/**
96  	 * Initializes GUI for this program.
97  	 */
98  	private void initGUI() {
99  		
100 		tabbedPane = new JTabbedPane();
101 		statusBar = new StatusBar(this);
102 		statusBar.initialize();
103 		tabbedPane.addChangeListener(new ChangeListener() {
104 
105 				@Override
106 				public void stateChanged(ChangeEvent e) {
107 					if(tabbedPane.getTabCount() == 0) {
108 						JNotepadPP.this.setTitle( "Notepad++" );
109 						statusBar.initialize();
110 						return;
111 					}
112 					Tab selectedTab = (Tab)tabbedPane.getSelectedComponent();
113 					if(selectedTab == null || selectedTab.getFileLocation() == null) {
114 						JNotepadPP.this.setTitle( "Unnamed.txt - Notepad++" );
115 						selectedTab.getEditor().getCaret().setDot(0);
116 						statusBar.initialize();
117 						return;
118 					}
119 					JNotepadPP.this.setTitle( selectedTab.getFileLocation().toString() + " - Notepad++");
120 					selectedTab.getEditor().getCaret().setDot(0);
121 					statusBar.initialize();
122 					return;
123 				}
124 			});
125 		
126 		getContentPane().setLayout(new BorderLayout());
127 		getContentPane().add(tabbedPane, BorderLayout.CENTER);
128 		getContentPane().add(statusBar, BorderLayout.SOUTH);
129 		
130 		createActions();
131 		createMenus();
132 		createToolbar();
133 	}
