85  	/**
86  	 * Creates new {@link Tab}.
87  	 * @param parent Parent pane in which this tab is.
88  	 * @param initialText initial text, if tab is opened from existing file. Null otherwise.
89  	 * @param initialFileLocation initial file location if tab is opened from existing file. Null otherwise.
90  	 * @param notepad Parent notepad containing this tab.
91  	 */
92  	public Tab(JTabbedPane parent, String initialText, Path initialFileLocation, JNotepadPP notepad) {
93  
94  		this.parent = parent;
95  		this.notepad = notepad;
96  		this.statusBar = notepad.getStatusBar();
97  		this.setLayout(new BorderLayout());
98  		this.editor = initialText == null ? new JTextArea() : new JTextArea(
99  				initialText);
100 		this.fileLocation = initialFileLocation;
101 
102 		this.editor.getDocument().addDocumentListener(new DocumentListener() {
103 
104 			private Document doc = editor.getDocument();
105 			
106 			@Override
107 			public void removeUpdate(DocumentEvent e) {
108 				setModified();
109 				statusBar.setLengthLabel(doc.getLength());
110 			}
111 
112 			@Override
113 			public void insertUpdate(DocumentEvent e) {
114 				setModified();
115 				statusBar.setLengthLabel(doc.getLength());
116 			}
117 
118 			@Override
119 			public void changedUpdate(DocumentEvent e) {
120 				setModified();
121 				statusBar.setLengthLabel(doc.getLength());
122 			}
123 		});
124 		this.editor.addCaretListener(new CaretListener() {
125 			
126 			private Document doc = editor.getDocument();
127 			
128 			@Override
129 			public void caretUpdate(CaretEvent e) {
130 				String text = "";
131 				try {
132 					text = doc.getText(0, editor.getCaret().getDot());
133 				} catch (BadLocationException ignorable) {
134 				}
135 				int column = 0;
136 				int line = 0;
137 				
138 				for(int i = 0; i < text.length(); i++) {
139 					char c = text.charAt(i);
140 					if( c == '\n') {
141 						line++;
142 					}
143 				}
144 				String lastLine = text;
145 				if(text.contains("\n")) {
146 					lastLine = text.substring(text.lastIndexOf('\n'));
147 				}
148 				column = lastLine.length();
149 				int selected = Math.abs(editor.getCaret().getDot() - editor.getCaret().getMark());
150 				
151 				if(selected != 0) {
152 					notepad.enableTextMenus();
153 				}
154 				else {
155 					notepad.disableTextMenus();
156 				}
157 				
158 				statusBar.setLineInfoLabel(line, column, selected);
159 			}
160 		});
161 		this.add(new JScrollPane(editor), BorderLayout.CENTER);
162 
163 		this.tabButton = new TabComponent(this);
164 		if (this.fileLocation != null) {
165 			this.tabButton.setToolTip(fileLocation.toString());
166 			this.tabButton.setTitle(fileLocation.getFileName().toString());
167 		}
168 		parent.add(this);
169 		parent.setTabComponentAt(parent.indexOfComponent(this), this.tabButton);
170 	}

236 	/**
237 	 * Tab component, below actual editor. Presents icon for modified and unmodified state,
238 	 * name of current document and close button.
239 	 * @author Vilim StaroveÅ¡ki
240 	 *
241 	 */
242 	private class TabComponent extends JPanel {
243 
244 		/**
245 		 * Default serial verision UID.
246 		 */
247 		private static final long serialVersionUID = 1L;
248 
249 		/**
250 		 * Icon for modified and unmodified state.
251 		 */
252 		private Icon icon;
253 		/**
254 		 * Name of document.
255 		 */
256 		private JLabel fileName;
257 		/**
258 		 * Close button, closes this document.
259 		 */
260 		private JButton closeButton;
261 
262 		/**
263 		 * Creates new {@link TabComponent}.
264 		 * @param tabParent tab that contains this {@link TabComponent}
265 		 */
266 		public TabComponent(Tab tabParent) {
267 			super(new FlowLayout());
268 			setOpaque(false);
269 			
270 			this.icon = greenDisc;
271 
272 			this.fileName = new JLabel("unnamed.txt");
273 			this.fileName.setIcon(icon);
274 			this.fileName.addMouseListener(new MouseAdapter() {
275 				
276 				@Override
277 				public void mouseClicked(MouseEvent e) {
278 					Tab.this.parent.setSelectedComponent(Tab.this);
279 				}
280 			});
281 				
282 
283 			this.closeButton = new JButton(cross);
284 			this.closeButton.setContentAreaFilled(false);
285 			this.closeButton.setFocusable(false);
286 			this.closeButton.setBorder(BorderFactory.createEtchedBorder());
287 			this.closeButton.setBorderPainted(false);
288 			this.closeButton.addMouseListener(new MouseListener() {
289 				
290 				@Override
291 				public void mouseReleased(MouseEvent e) {
292 				}
293 				
294 				@Override
295 				public void mousePressed(MouseEvent e) {
296 				}
297 				
298 				@Override
299 				public void mouseExited(MouseEvent e) {
300 					closeButton.setBorderPainted(false);
301 				}
302 				
303 				@Override
304 				public void mouseEntered(MouseEvent e) {
305 					closeButton.setBorderPainted(true);
306 				}
307 				
308 				@Override
309 				public void mouseClicked(MouseEvent e) {
310 					int i = parent.indexOfTabComponent(TabComponent.this);
311 					if(tabParent.modified) {
312 						int r = JOptionPane.showConfirmDialog(parent, 
313 								"'"+TabComponent.this.fileName.getText()+
314 								"' has unsaved changes. Do you want to save it?", 
315 								"Warning", 
316 								JOptionPane.YES_NO_OPTION,
317 								JOptionPane.QUESTION_MESSAGE);
318 						if(r == JOptionPane.YES_OPTION) {
319 							Tab.this.saveDoc();
320 						}
321 					}
322 					if (i != -1) {
323 						parent.remove(i);
324 					}
325 				}
326 			});
327 			this.closeButton.setRolloverEnabled(true);
328 
329 			add(fileName);
330 			add(closeButton);
331 		}
332
